cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.SolidMechanics.FEM.Damping LANGUAGES CXX)

set(SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR "src/sofa/component/solidmechanics/fem/damping/")

set(HEADER_FILES
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/init.h
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/DiagonalVelocityDampingForceField.h
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/DiagonalVelocityDampingForceField.inl
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/UniformVelocityDampingForceField.h
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/UniformVelocityDampingForceField.inl
)

set(SOURCE_FILES
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/DiagonalVelocityDampingForceField.cpp
    ${SOFACOMPONENTSOLIDMECHANICSFEMDAMPING_SOURCE_DIR}/UniformVelocityDampingForceField.cpp
)

sofa_find_package(Sofa.SimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC Sofa.SimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
